SELECT CONCAT(NAME,CONCAT('(',CONCAT(substr(OCCUPATION,1,1),')'))) FROM OCCUPATIONS ORDER BY NAME;

SELECT CONCAT('There are a total of ', COUNT(OCCUPATION)), CASE WHEN COUNT(OCCUPATION) > 1 THEN LOWER(CONCAT(OCCUPATION,'s.')) ELSE OCCUPATION END FROM OCCUPATIONS GROUP BY OCCUPATION ORDER BY 1 ASC, 2 ASC;
